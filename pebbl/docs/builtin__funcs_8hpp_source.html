43inline PEBBLObject length_impl(const std::vector<PEBBLObject>& args, class Interpreter& interp) {
44 if (args.size() != 1) {
45 interp.report_error("length() expects exactly 1 argument, got " + std::to_string(args.size()));
46 return PEBBLObject::make_null();
47 }
48
49 const auto& obj = args[0];
50 if (obj.is_gc_ptr()) {
51 auto* gc_obj = obj.as_gc_ptr();
52 switch (gc_obj->tag) {
53 case GCTag::STRING: {
54 auto* str = static_cast<PEBBLString*>(gc_obj);
55 return PEBBLObject::make_int32(static_cast<int32_t>(str->length()));
56 }
57 case GCTag::ARRAY: {
58 auto* arr = static_cast<PEBBLArray*>(gc_obj);
59 return PEBBLObject::make_int32(static_cast<int32_t>(arr->length()));
60 }
61 case GCTag::DICT: {
62 auto* dict = static_cast<PEBBLDict*>(gc_obj);
63 return PEBBLObject::make_int32(static_cast<int32_t>(dict->size()));
64 }
65 default:
66 break;
67 }
68 }
69 interp.report_error("length() can only be called on strings, arrays, or dictionaries");
70 return PEBBLObject::make_null();

94 } else if (obj.is_double()) {
95 type_name = "float";

97 auto* gc_obj = obj.as_gc_ptr();
98 switch (gc_obj->tag) {
99 case GCTag::STRING:
100 type_name = "string";
101 break;
102 case GCTag::ARRAY:
103 type_name = "array";
104 break;
105 case GCTag::DICT:
106 type_name = "dict";

112 type_name = "builtin_function";

117 }

119 type_name = "unknown";
